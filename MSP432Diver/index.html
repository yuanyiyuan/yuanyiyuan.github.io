    看门狗模式： 定时器将在计数到期时复位 ，复位可以设置为软复位或硬复位
        eg:
        MAP_SysCtl_setWDTTimeoutResetType ( SYSCTL_SOFT_RESET );    //设置为软复位
        MAP_WDT_A_initWatchdogTimer ( WDT_A_CLOCKSOURCE_SMCLK ,     //设置来源为SMCLK
                                    WDT_A_CLOCKITERATIONS_512K )  //时钟更新为512000个SMCLK周期  【若运行512000个周期后没有更新，则复位】
    间隔模式： 这将 WDT 变成了一个具有中断支持的普通 16 位递减计数器。如果用户需要访问另一个低功耗计数器，但已经使用了其他资源，则可以使用此选项
        eg:
        MAP_WDT_A_initIntervalTimer ( WDT_A_CLOCKSOURCE_VLOCLK ,    //设置来源为SMCLK
                                    WDT_A_CLOCKITERATIONS_32K );  //时钟更新为32000个SMCLK周期
    ································································

                                    uint_fast8_t     mask )                   //中断源
                                            |--> EUSCI_A_UART_RECEIVE_INTERRUPT -接收中断
                                                EUSCI_A_UART_TRANSMIT_INTERRUPT - 传输中断
                                                EUSCI_A_UART_RECEIVE_ERRONEOUSCHAR_INTERRUPT - 接收错误字符中断使能
                                                EUSCI_A_UART_BREAKCHAR_INTERRUPT - 接收中断字符中断使能
    禁用单个 UART 中断源： （参数同上）                                            
    void UART_disableInterrupt    (    uint32_t     moduleInstance,
                                    uint_fast8_t     mask )       
    清除 UART 中断源： （UART 中断源被清除，使其不再有效，使用中断向量发生器时，最高中断标志会自动清除）
    void UART_clearInterruptFlag    (    uint32_t     moduleInstance,
                                        uint_fast8_t     mask )    
    为 UART 中断注册一个中断处理程序：
    void UART_registerInterrupt    (    uint32_t     moduleInstance,       //实例: EUSCI_Ax_BASE [x 为0~3]
                                void(*)(void)     intHandler)     //处理程序（指针），指向发生定时器捕捉比较中断时要调用的函数                              
    取消注册 UART 模块的中断处理程序：
    void UART_unregisterInterrupt    (    uint32_t     moduleInstance    )    //实例: EUSCI_Ax_BASE [x 为0~3]

ADC14

ADC14:
    启用 ADC 模块：
    void ADC14_enableModule    (void)    
    禁用 ADC 模块：
    bool AD​​C14_disableModule (void)
    初始化 ADC 模块并设置时钟系统分频器/预分频器：
    bool ADC14_initModule    (    uint32_t     clockSource,               //时钟源  ADC_CLOCKSOURCE_ADCOSC [默认] SYSOSC ACLK MCLK
                                uint32_t     clockPredivider,           //时钟分频器  ADC_PREDIVIDER_1 [默认]  4  32  64
                                uint32_t     clockDivider,              //时钟分频器  ADC_DIVIDER_1 [默认值] 1~8
                                uint32_t     internalChannelMask )       //内部通道掩码  ADC_MAPINTCH3 2 1 0...
                                    |--> 配置 ADC 模块的内部/外部引脚映射 设置确定给定的 ADC 通道或组件是映射到外部引脚（默认），还是路由到内部组件
    设置 ADC 模块的转换率:                                
    void ADC14_setResolution    (    uint32_t     resolution    )          //转换率 
                                                                            ADC_8BIT（10个时钟周期转换时间）
                                                                            ADC_10BIT（12个时钟周期转换时间）
                                                                            ADC_12BIT（14个时钟周期转换时间）
                                                                            ADC_14BIT（16 个时钟周期转换时间）[DEFAULT]
    获取 ADC 模块的转换率:                                           
    uint_fast32_t ADC14_getResolution    (void)            //转换率   
    设置 ADC 模块的触发源:
    bool ADC14_setSampleHoldTrigger    (    uint32_t     source,        //采样触发源 ADC_TRIGGER_ADCSC [默认], ADC_TRIGGER_SOURCE1  1~7
                                        bool     invertSignal )       //触发方式，true使用下降沿，false使用上升沿

    将 ADC 模块配置为使用多存储器采样：
    【多个样本将连续发生并存储在多个内存位置。第一个样本/转换将放置在 memoryStart 中，而最后一个样本将存储在 memoryEnd 中】
    bool ADC14_configureMultiSequenceMode    (    uint32_t     memoryStart,     //存储第一个样本/转换值的内存位置
                                                uint32_t     memoryEnd,       //存储最后一个样本的内存位置
                                                bool     repeatMode )     //指定在第一轮采样/转换后是否重复转换/采样周期                               
    将 ADC 模块配置为使用单个 ADC 存储器位置进行采样/转换：

    bool ADC14_configureSingleSampleMode    (    uint32_t     memoryDestination,       //存储样本/转换值的内存位置
                                                bool     repeatMode )                 //指定在第一轮采样/转换后是否重复转换/采样周期
    启用 ADC 数据的转换：
    bool ADC14_enableConversion     (void)    
    停止 ADC 模块的转换转换：
    void ADC14_disableConversion  (void)    

    为 ADC 中断注册一个中断处理程序：
    void ADC14_registerInterrupt    (void(*)(void)     intHandler)    
    取消注册 ADCC 模块的中断处理程序：
    void ADC14_unregisterInterrupt    (void)
